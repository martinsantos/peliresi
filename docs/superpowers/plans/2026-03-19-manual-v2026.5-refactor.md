```javascript
// ── v2026.5: Theme toggle ──
function initTheme() {
  var saved = localStorage.getItem('sitrep-manual-theme');
  if (saved) {
    document.documentElement.setAttribute('data-theme', saved);
  }
}
initTheme(); // Run immediately to avoid flash

function toggleTheme() {
  var current = document.documentElement.getAttribute('data-theme');
  var next = current === 'dark' ? 'light' : (current === 'light' ? 'dark' :
    (window.matchMedia('(prefers-color-scheme: dark)').matches ? 'light' : 'dark'));
  document.documentElement.setAttribute('data-theme', next);
  localStorage.setItem('sitrep-manual-theme', next);
  var btn = document.getElementById('themeToggle');
  if (btn) btn.textContent = next === 'dark' ? '\u2600\uFE0F' : '\uD83C\uDF19';
}

// ── v2026.5: Reading progress bar ──
function updateProgress() {
  var bar = document.getElementById('progressBar');
  if (!bar) return;
  var scrollTop = window.scrollY;
  var docHeight = document.documentElement.scrollHeight - window.innerHeight;
  var pct = docHeight > 0 ? (scrollTop / docHeight) * 100 : 0;
  bar.style.width = pct + '%';
}
```

Add `updateProgress()` call inside the existing scroll event listener.

## Risk Mitigations

| Risk | Mitigation |
|------|------------|
| CSS extraction breaks layout | Visual check after Step 1.3 before continuing |
| JS extraction breaks interactivity | Manual test after Step 2.5 |
| swagger-jsdoc annotations cause build errors | Build + verify after each route file |
| Swagger UI too heavy for production | swagger-ui-express serves pre-built static assets, no runtime cost |
| Dark mode CSS conflicts | Uses CSS custom properties + `[data-theme]` selector, isolated from app |
| Print stylesheet breaks | Test Ctrl+P after Step 1.3 |
